async function handleScanResult(rawUrl) { const phone = getCurrentPhone(); if (!isValidMobile(phone)) { await stopScanner(true); statusEl.textContent = "無效的手機號碼。"; return; } let urlObj, mode, plantId, spotId, picNum; try { urlObj = new URL(rawUrl.trim()); mode = urlObj.searchParams.get("mode"); plantId = urlObj.searchParams.get("plant"); spotId = urlObj.searchParams.get("spot"); picNum = urlObj.searchParams.get("pic"); } catch (err) { return; } if (mode === "ar" && picNum) { if (navigator.vibrate) navigator.vibrate(100); await stopScanner(true); openAr(picNum); return; } if (mode !== "plant_go" && mode !== "stamp") return; if (navigator.vibrate) navigator.vibrate([100, 50, 100]); // 瞬間顯示處理中 document.getElementById("loader-text").innerHTML = "正在存入點數...
(請稍候約 2 秒鐘)"; document.getElementById("global-loader").style.display = "flex"; await stopScanner(true); document.getElementById("scanActions").style.display = "none"; try { if (mode === "plant_go" && plantId) { // 呼叫 GAS 記點 let resp; try { resp = await jsonpMarkPlant(plantId, phone); } catch(e) { resp = await jsonpMarkPlant(plantId, phone); } const target = resp.redirectUrl || COLLECTION_URL; // ★ 記點成功！不留戀，瞬間把任務交給 plant.html 中繼站處理 const msg = encodeURIComponent(resp.message || "已完成記點"); const safeUrl = encodeURIComponent(target); window.location.replace(`plant.html?msg=${msg}&url=${safeUrl}`); } else if (mode === "stamp" && spotId) { // 如果是掃描點位，直接交給 GAS 處理 (GAS 已經寫好會轉回 scanner) urlObj.searchParams.set("phone", phone); window.location.replace(urlObj.toString()); } } catch (err) { document.getElementById("global-loader").style.display = "none"; statusEl.innerHTML = "系統連線失敗 💦
請點擊下方按鈕重新啟動相機。"; statusEl.style.background = "#f8d7da"; document.getElementById("scanActions").style.display = "grid"; document.getElementById("startBtn").style.display = "block"; } }
